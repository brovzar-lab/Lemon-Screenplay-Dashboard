    export function getPendingWriteCount(): number {
      try {
        const raw = localStorage.getItem(PENDING_QUEUE_KEY);
        if (!raw) return 0;
        const queue = JSON.parse(raw);
        return Array.isArray(queue) ? queue.length : 0;
      } catch {
        return 0;
      }
    }
